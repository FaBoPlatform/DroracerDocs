#include "us_ticker_api.h"
#include "SPI.h"

void setup() {
  pinMode(2, OUTPUT);
  pinMode(3, OUTPUT);
  pinMode(4, OUTPUT);
  pinMode(A0, INPUT);
  pinMode(A3, INPUT);
  Serial.begin(115200);
  SPI.begin();
  us_ticker_init();
  Serial.println("PLSEIN_JetRacer_Start");
}


void startBit() {
  byte start = 0x00;
  SPI.transfer(start);
  SPI.transfer(start);
  SPI.transfer(start);
  SPI.transfer(start);
}

void endBit() {
  byte end = 0x00;
  SPI.transfer(end);
  SPI.transfer(end);
  SPI.transfer(end);
  SPI.transfer(end);
}

void setRGB(short r, short g, short b) {
  SPI.transfer(0xEF);
  SPI.transfer(r);
  SPI.transfer(g);
  SPI.transfer(b);
}


uint32_t pulsein(void){
  while (!digitalRead(A3)); // wait for high
    uint32_t times_start = (us_ticker_read)();
  while (digitalRead(A3)); // wait for low
    uint32_t times_end =  (us_ticker_read)();
    Serial.println(times_end-times_start);
  return times_end-times_start;
  }

void loop() {

    uint32_t duration = pulsein();
     if(duration > 1100){
    //digitalWrite(2, HIGH);
    //digitalWrite(3, HIGH);
    digitalWrite(4, LOW);
    startBit();

    setRGB(80,   0,   10);
    setRGB(80,   0,   40);
    setRGB(80,   0,   90);
    setRGB(80,   0,   130);
    setRGB(80,   0,   170);
    setRGB(80,   0,   210);
    setRGB(80,  0,   255);
    endBit();
  } else{
    //digitalWrite(2, LOW);
    //digitalWrite(3, LOW);
    digitalWrite(4, HIGH);
    startBit();
    setRGB(0,   255,   0);
    setRGB(0,   255,   0);
    setRGB(0,   255,   0);
    setRGB(0,   255,   0);
    setRGB(0,   255,   0);
    setRGB(0,   255,   0);
    endBit();

  } 
}
